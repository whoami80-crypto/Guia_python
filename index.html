Listas

Las listas son colecciones ordenadas y mutables de elementos.

# Crear una lista
frutas = ["manzana", "banana", "cereza"]
numeros = [1, 2, 3, 4, 5]

# Acceder a elementos
print(frutas[0]) # "manzana"
print(frutas[-1]) # "cereza" (último elemento)

# Modificar elementos
frutas[1] = "naranja"

# Añadir elementos
frutas.append("uva") # Añade al final
frutas.insert(1, "mango") # Inserta en posición específica

# Eliminar elementos
frutas.remove("manzana") # Elimina por valor
frutas.pop(0) # Elimina por índice

# Métodos útiles
longitud = len(frutas)
frutas.sort() # Ordena la lista
frutas.reverse() # Invierte el orden

Tuplas

Las tuplas son colecciones ordenadas e inmutables de elementos.

# Crear una tupla
coordenadas = (10, 20)
colores = ("rojo", "verde", "azul")

# Acceder a elementos
print(coordenadas[0]) # 10
print(colores[1:3]) # ("verde", "azul") - slicing

# Las tuplas son inmutables
# coordenadas[0] = 15 # Esto daría error

# Desempaquetado de tuplas
x, y = coordenadas
print(f"x: {x}, y: {y}") # x: 10, y: 20

Diccionarios

Los diccionarios son colecciones no ordenadas de pares clave-valor.

# Crear un diccionario
persona = {
"nombre": "Ana",
"edad": 30,
"ciudad": "Madrid"
}

# Acceder a valores
print(persona["nombre"]) # "Ana"
print(persona.get("edad")) # 30

# Modificar valores
persona["edad"] = 31

# Añadir nuevos pares clave-valor
persona["profesion"] = "Ingeniera"

# Eliminar elementos
del persona["ciudad"]
profesion = persona.pop("profesion")

# Métodos útiles
claves = persona.keys()
valores = persona.values()
items = persona.items()

# Recorrer un diccionario
for clave, valor in persona.items():
print(f"{clave}: {valor}")

Conjuntos

Los conjuntos son colecciones no ordenadas de elementos únicos.

# Crear un conjunto
numeros = {1, 2, 3, 4, 5}
letras = set(["a", "b", "c", "a"]) # {"a", "b", "c"}

# Añadir elementos
numeros.add(6)

# Eliminar elementos
numeros.remove(3)
numeros.discard(10) # No da error si el elemento no existe

# Operaciones con conjuntos
A = {1, 2, 3}
B = {3, 4, 5}

union = A | B # {1, 2, 3, 4, 5}
interseccion = A & B # {3}
diferencia = A - B # {1, 2}
diferencia_simetrica = A ^ B # {1, 2, 4, 5}

Funciones en Python

Las funciones permiten organizar el código en bloques reutilizables.

Definición de funciones

# Función simple
def saludar():
print("¡Hola!")

# Función con parámetros
def saludar_persona(nombre):
print(f"¡Hola, {nombre}!")

# Función con retorno
def sumar(a, b):
return a + b

# Función con parámetros por defecto
def presentar(nombre, edad=30):
print(f"Soy {nombre} y tengo {edad} años.")

# Llamadas a funciones
saludar()
saludar_persona("Ana")
resultado = sumar(5, 3)
presentar("Carlos")
presentar("Laura", 25)

Funciones lambda (anónimas)

# Función lambda simple
cuadrado = lambda x: x ** 2
print(cuadrado(5)) # 25

# Uso con map
numeros = [1, 2, 3, 4, 5]
cuadrados = list(map(lambda x: x ** 2, numeros))
print(cuadrados) # [1, 4, 9, 16, 25]

# Uso con filter
pares = list(filter(lambda x: x % 2 == 0, numeros))
print(pares) # [2, 4]

Ejercicio: Calculadora de Factorial

Crea una función que calcule el factorial de un número (n!) de forma recursiva.

def factorial(n):
if n == 0 or n == 1:
return 1
else:
return n * factorial(n - 1)

Bucles y Condiciones

Condicionales: if, elif, else

edad = 18

if edad < 18:
print("Eres menor de edad")
elif edad == 18:
print("Tienes exactamente 18 años")
else:
print("Eres mayor de edad")

# Condicionales anidados
nota = 85

if nota >= 90:
print("A")
elif nota >= 80:
print("B")
elif nota >= 70:
print("C")
else:
print("F")

Bucles: for y while

# Bucle for
for i in range(5):
print(i) # 0, 1, 2, 3, 4

frutas = ["manzana", "banana", "cereza"]
for fruta in frutas:
print(fruta)

# Bucle while
contador = 0
while contador < 5:
print(contador)
contador += 1

# Control de bucles: break y continue
for i in range(10):
if i == 3:
continue # Salta esta iteración
if i == 7:
break # Sale del bucle
print(i) # 0, 1, 2, 4, 5, 6

Programación Orientada a Objetos

La POO permite modelar entidades del mundo real mediante clases y objetos.

Clases y Objetos

# Definición de una clase
class Persona:
# Constructor
def __init__(self, nombre, edad):
self.nombre = nombre
self.edad = edad

# Método de instancia
def presentarse(self):
print(f"Hola, soy {self.nombre} y tengo {self.edad} años.")

# Método para modificar atributos
def cumplir_anios(self):
self.edad += 1
print(f"¡Ahora tengo {self.edad} años!")

# Crear objetos (instancias)
persona1 = Persona("Ana", 25)
persona2 = Persona("Carlos", 30)

# Usar métodos
persona1.presentarse()
persona2.presentarse()
persona1.cumplir_anios()

Herencia

# Clase base
class Animal:
def __init__(self, nombre):
self.nombre = nombre

def hacer_sonido(self):
pass # Método abstracto

# Clases derivadas
class Perro(Animal):
def hacer_sonido(self):
return "¡Guau!"

class Gato(Animal):
def hacer_sonido(self):
return "¡Miau!"

# Uso de la herencia
perro = Perro("Rex")
gato = Gato("Whiskers")

print(f"{perro.nombre} dice: {perro.hacer_sonido()}")
print(f"{gato.nombre} dice: {gato.hacer_sonido()}")

Encapsulación

class CuentaBancaria:
def __init__(self, titular, saldo_inicial=0):
self.titular = titular
self.__saldo = saldo_inicial # Atributo privado

# Métodos públicos para acceder al saldo
def depositar(self, cantidad):
if cantidad > 0:
self.__saldo += cantidad
print(f"Depósito exitoso. Nuevo saldo: {self.__saldo}")
else:
print("La cantidad debe ser positiva")

def retirar(self, cantidad):
if 0 < cantidad <= self.__saldo:
self.__saldo -= cantidad
print(f"Retiro exitoso. Nuevo saldo: {self.__saldo}")
else:
print("Fondos insuficientes o cantidad inválida")

def consultar_saldo(self):
return self.__saldo

# Uso de la clase
cuenta = CuentaBancaria("Ana", 1000)
cuenta.depositar(500)
cuenta.retirar(200)
print(f"Saldo actual: {cuenta.consultar_saldo()}")

Ejercicio: Sistema de Gestión de Vehículos

Crea un sistema de gestión de vehículos con una clase base Vehiculo y clases derivadas Coche y Moto. Cada vehículo debe tener marca, modelo y año, y un método para mostrar información.

class Vehiculo:
def __init__(self, marca, modelo, año):
self.marca = marca
self.modelo = modelo
self.año = año

def mostrar_info(self):
return f"{self.marca} {self.modelo} ({self.año})"

class Coche(Vehiculo):
def __init__(self, marca, modelo, año, num_puertas):
super().__init__(marca, modelo, año)
self.num_puertas = num_puertas

def mostrar_info(self):
return f"Coche: {super().mostrar_info()}, {self.num_puertas} puertas"

class Moto(Vehiculo):
def __init__(self, marca, modelo, año, tipo):
super().__init__(marca, modelo, año)
self.tipo = tipo

def mostrar_info(self):
return f"Moto: {super().mostrar_info()}, tipo: {self.tipo}"

# Crear vehículos
coche = Coche("Toyota", "Corolla", 2020, 4)
moto = Moto("Honda", "CBR", 2021, "deportiva")

print(coche.mostrar_info())
print(moto.mostrar_info())
